class point():
    def __init__(self, x, y):
        self.x = x
        self.y = y

    def s(self, x, y):#set
        self.x = x
        self.y = y

    def g(self):#get
        return (self.x, self.y)

    def m(self, x1, y1):#move
        self.x = self.x+x1
        self.y = self.y+y1
a=point(1,1)
a.s(2,2)
print(a.g())
a.m(3,3)
print(a.g())
